	СуммаЧисел = СуммаЧисел + Счётчик;

	МассивЧисел.Добавить(Счётчик);

//Возведение в степень через цикл

Если Результат > 10000 Тогда
	Сообщить("Результат превышает допустимую норму в 10000, " + Результат);  
КонецЕсли;
КонецЦикла;
Сообщить(Результат);
